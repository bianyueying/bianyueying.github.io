接口具体实现：这样就能返回树形的数据了

package bml.service.impl;

@Service
public class LabelTreeServiceImpl implements LabelTreeService {

    @Resource
    EletreePlusMapper mapper;

    @Override
    public List<EleTreeDTO> getPermsTree() {
        //原始的数据
        List<EleTreeDTO> before = mapper.menuList();
        //最后的结果
        List<EleTreeDTO> after = new ArrayList<>();
        for (EleTreeDTO bml: before) {
            if (bml.getParentId() == 0) {
                after.add(bml);
            }
        }
        for (EleTreeDTO bml : after) {
            List<EleTreeDTO> childrenList = getChild(bml.getId().intValue(), before);
            bml.setChildren(childrenList);
        }
        return after;
    }

    /**
     * 网上找的Java树形结构工具类 一知半解 但它能给我正确结果就足够了
     */
    private List<EleTreeDTO> getChild(Integer id, List<EleTreeDTO> list) {
        List<EleTreeDTO> childrenList = new ArrayList<>();
        for (EleTreeDTO eleTreeDTO : list) {
            if (eleTreeDTO.getParentId().equals(id)) {
                childrenList.add(eleTreeDTO);
            }
        }
        for (EleTreeDTO eleTreeDTO : childrenList) {
            eleTreeDTO.setChildren(getChild(eleTreeDTO.getId().intValue(),list));
        }
        if (childrenList.size() == 0) {
            return null;
        }
        return childrenList;
    }
}

前台EleTree通过使用Ajax获取到相关数据并完成初始化，其他例如 添加子节点、删除节点 等一系列操作也都是按照EleTree相关文档完成，只需要改变一下URL，其它再完善亿点点细节行了，这里就不再赘述…

11.后端实现

package bml.controller;

@RestController
public class EletreeController {

    @Resource
    EletreeService service;

    @ApiOperation("修改节点")
    @PutMapping("/user/tree")
    public BmlResult<Object> updateTree(@RequestBody Eletree eletree) {
        try {
            service.updateById(eletree);
            return new BmlResult<>(200,"修改成功！");
        } catch (Exception e) {
            return new BmlResult<>(400,"修改失败...");
        }
    }

    @ApiOperation("根据id删除节点")
    @DeleteMapping("/user/tree")
    public BmlResult<Object> deleteTree(String id) {
        try {
            // 此处需要根据实体类的ID属性进行对应转换
            service.removeById(Long.parseLong(id));
            return new BmlResult<>(200,"删除成功！");
        } catch (Exception e) {
            return new BmlResult<>(400,"删除失败...");
        }
    }

    @ApiOperation("添加节点")
    @PostMapping("/user/tree")
    public BmlResult<Object> addTree(@RequestBody Eletree eletree) {
        try {
            service.save(eletree);
            return new BmlResult<>(200,"添加成功！");
        } catch (Exception e) {
            return new BmlResult<>(400,"添加失败...");
        }
    }
}
