public class GeneratorCode {
    public static void main(String[] args) {
        AutoGenerator autoGenerator = new AutoGenerator();
        //全局配置
        GlobalConfig config = new GlobalConfig();
        String projectPath = System.getProperty("user.dir");
        config.setOutputDir(projectPath+"/src/main/java");
        config.setAuthor("XX");
        config.setOpen(false);
        //是否覆盖
        config.setFileOverride(true);
        config.setServiceName("%sService");
        config.setIdType(IdType.AUTO);
        config.setDateType(DateType.ONLY_DATE);
        config.setSwagger2(true);
        autoGenerator.setGlobalConfig(config);

        //数据源配置
        DataSourceConfig sourceConfig = new DataSourceConfig();
        sourceConfig.setUrl("jdbc:mysql://127.0.0.1:3306/springboot?useUnicode=true&characterEncoding=utf8&serverTimezone=GMT%2B8&useSSL=false");
        sourceConfig.setDriverName("com.mysql.cj.jdbc.Driver");
        sourceConfig.setUsername("root");
        sourceConfig.setPassword("123456");
        sourceConfig.setDbType(DbType.MYSQL);
        autoGenerator.setDataSource(sourceConfig);

        //包配置
        PackageConfig packageConfig = new PackageConfig();
        packageConfig.setParent("bml");
        packageConfig.setEntity("entity");
        packageConfig.setMapper("mapper");
        packageConfig.setService("service");
        packageConfig.setController("controller");
        autoGenerator.setPackageInfo(packageConfig);

        // 策略配置
        StrategyConfig strategy = new StrategyConfig();
        //设置要映射的表名
        strategy.setInclude("echarts");
        strategy.setNaming(NamingStrategy.underline_to_camel);
        strategy.setColumnNaming(NamingStrategy.underline_to_camel);
        strategy.setEntityLombokModel(true);
        strategy.setLogicDeleteFieldName("deleted");

        //自动填充配置
        TableFill createTime = new TableFill("create_time", FieldFill.INSERT);
        TableFill updateTime = new TableFill("update_Time", FieldFill.INSERT_UPDATE);
        ArrayList<TableFill> list = new ArrayList<>();
        list.add(createTime);
        list.add(updateTime);
        strategy.setTableFillList(list);
        strategy.setRestControllerStyle(true);
        autoGenerator.setStrategy(strategy);
        autoGenerator.execute();
    }
}

    List<Echarts> listAll();

8.接口方法实现

@Service
public class EchartsServiceImpl extends ServiceImpl<EchartsMapper, Echarts> implements EchartsService {

    @Resource
    EchartsMapper mapper;

    @Override
    public List<Echarts> listAll() {
        QueryWrapper<Echarts> wrapper = new QueryWrapper<>();
        // 降序排列
        wrapper.orderByDesc("id");
        byte maxLatestCount=7;
        // 查询出全部结果
        List<Echarts> list = mapper.selectList(wrapper);
        // 如果结果集大于7个 则返回最新的7个数据;如果小于7个就返回全部数据
        if (list.size() > maxLatestCount) {
            ArrayList<Echarts> list1 = new ArrayList<>();
            for (int i = 0; i < maxLatestCount; i++) {
                list1.add(list.get(i));
            }
            return list1;
        }
        return list;
    }
}

9.控制层调用

控制层主要是调用服务层，获取到相关数据并返回给前台对应的接口

@RestController
public class EchartsController {

    @Resource
    EchartsService echartsService;

    @RequestMapping(value = "/admin/list",method = RequestMethod.GET)
    public List<Echarts> getList() {
        List<Echarts> list = echartsService.listAll();
        // 反转集合中的元素 让最新的在后面
        Collections.reverse(list);
        // 测试是否能获取到正确数据 成功后删除该循环
        for (Echarts value : list) {
            System.out.println(value);
        }
        return list;
    }
}
