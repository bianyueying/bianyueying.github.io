SpringBoot + Editor.md

最近学习Layui的过程中要使用到富文本编辑器，不过看Layui自带的好像不太好用，同时Editor名头不小，所有就写个小例子试一下吧！

快速开始

1.创建数据库

核心字段主要是id、description，其它可有可无；创建后手动输入几条数据

CREATE TABLE `bml_markdown` (
  `id` INT NOT NULL AUTO_INCREMENT COMMENT '主键',
  `description` TEXT COMMENT '文本内容',
  `deleted` INT DEFAULT '0' COMMENT '逻辑删除 默认为0,1为删除',
  `create_time` DATETIME DEFAULT NULL COMMENT '添加时间',
  `update_time` DATETIME DEFAULT NULL COMMENT '更改时间',
  PRIMARY KEY (`id`)
) ENGINE=INNODB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='富文本测试表'

        /**
     * 查询所有数据并返回
     * @param page 默认自带
     * @param limit 默认自带
     */
    @ResponseBody
    @GetMapping("/admin/markdown")
    public BmlResult<Object> list(@RequestParam("page") Integer page, @RequestParam("limit") Integer limit) {
        List<BmlMarkdown> markdowns = markdownService.listAll(page, limit);
        // HashMap采用key-value格式 允许一个key为null 多个value为null
        Map<String,Object> result = new HashMap<>(100);
        result.put("count",markdownService.count(null));
        result.put("data",markdowns);
        return new BmlResult<>(result,0);
    }

    /**
     * 根据id查询出文本内容并返回 用于回显
     */
    @ResponseBody
    @GetMapping("/admin/markdown/{id}")
    public String getMarkdown(@PathVariable Integer id) {
        return markdownService.getById(id).getDescription();
    }

    /**
     * 根据id删除
     */
    @ResponseBody
    @DeleteMapping("/admin/markdown/{id}")
    public BmlResult<Object> delete(@PathVariable Integer id) {
        try {
            markdownService.removeById(id);
            return new BmlResult<>(200,"删除成功！");
        } catch (Exception e) {
            return new BmlResult<>(400,"删除失败...");
        }
    }

    /**
     * 更新文本 注意可以不用接收id参数
     */
    @ResponseBody
    @PutMapping("/admin/markdown/")
    public BmlResult<Object> update(@RequestBody BmlMarkdown markdown) {
        try {
            markdownService.updateById(markdown);
            return new BmlResult<>(200,"更新成功！");
        } catch (Exception e) {
            return new BmlResult<>(400,"更新失败...");
        }
    }

10.总结

总的来说，富文本编辑器的使用在一些后台管理系统中也是比较普遍的，适合一些博客文章的编写等。在该例子中，我使用了市面上常用的Editor.md集成Layui，而没有使用Layui自带的编辑器（仅看工具栏好像没有E强大，反正也没用过…），整个过程中，感觉文本的编辑回显那部分稍显麻烦，文本的获取需要重新请求后台，然后再 放 到编辑器里面，第一次使用的时候也走了一点弯路，不过还好；其他就没什么需要特别注意的点了，按照正常思路完全可以做下来，就酱吧！睡觉睡觉
